CTPShortName = {
    type = character
    text = {
        trigger ={
            is_local_player = yes
        }
        localization_key = CTP_Me
    }
    text = {
		trigger = {
			arabic_check = yes
		}
		localization_key = CTP_Short_Name_Arabic
	}
    text = {
		trigger = {
			persian_standard_check = yes
		}
		localization_key = CTP_Short_Name_Persian
    }
    text = {
		trigger = {
			culture = {
                is_cnp_supported_culture = yes
                has_cultural_pillar = language_norse
            }
		}
		localization_key = CTP_Short_Name_North
	}
	text = {
		fallback = yes
		localization_key = CTP_Short_Name_Default
	}
}
