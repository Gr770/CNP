    type = character
    text = {
		trigger = {
			exists = father
            father = {
                OR = {
                    has_character_flag = name_pack_support
                    AND = {
                        exists = dynasty
                        dynasty = {
                            any_in_list = {
                                variable = name_pack_support_dynasty
                                this = root.father
                            }
                        }
                    }
                    any_in_global_list = {
                        variable = name_pack_support_global
                        this = root.father
                    }
                }
            }
            is_lowborn = no
            is_independent_ruler = no
            NOT = {
                highest_held_title_tier > tier_duchy
                father = {
                    highest_held_title_tier > tier_duchy
                }
                AND = {
                    exists = mother
                    mother = {
                        highest_held_title_tier > tier_duchy
                    }
                }
                AND = {
                    exists = primary_spouse
                    primary_spouse = {
                        is_landed = yes
                    }
                }
            }
		}
		localization_key = IA_Patronym
	}
    text = {
		trigger = {
