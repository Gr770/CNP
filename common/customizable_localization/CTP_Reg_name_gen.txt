CTPRegName = {
    type = character
	text = {
		trigger = {
			arabic_check = yes
		}
		localization_key = CTP_Name_Arabic
	}
    text = {
		trigger = {
			persian_standard_check = yes
		}
		localization_key = CTP_Name_Persian
    }
	text = {
		trigger = {
			culture = {
                is_cnp_supported_culture = yes
                has_cultural_pillar = language_norse
            }
		}
		localization_key = CTP_Name_North
	}
	text = {
		fallback = yes
		localization_key = CTP_Reg_Name_Default
	}
}
