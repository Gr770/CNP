CTPTierCheck = {
    type = character
    text = {
        trigger = {
            OR = {
                is_ruler = yes
                AND = {
                    exists = father
                    father = {
                        highest_held_title_tier >= tier_kingdom
                        OR = {
                            has_government = feudal_government
                            has_government = clan_government
                        }
                    }
                }
                AND = {
                    exists = mother
                    mother = {
                        highest_held_title_tier >= tier_kingdom
                        OR = {
                            has_government = feudal_government
                            has_government = clan_government
                        }
                    }
                }
                AND = {
                    is_male = no
                    any_child = {
                        highest_held_title_tier >= tier_kingdom
                        OR = {
                            has_government = feudal_government
                            has_government = clan_government
                        }
                    }
                }
            }
        }
        localization_key = CTP_tier_checked
    }
	text = {
		fallback = yes
		localization_key = Gui_blank
	}
}

CTPNameCheck = {
    type = character
    text = {
        trigger = {
            OR = {
                has_character_flag = name_pack_support
                AND = {
                    exists = dynasty
                    dynasty = {
                        any_in_list = {
                            variable = name_pack_support_dynasty
                            this = root
                        }
                    }
                }
                any_in_global_list = {
                    variable = name_pack_support_global
                    this = root
                }
            }
        }
        localization_key = CTP_supported_name
    }
    text = {
        fallback = yes
		localization_key = CTP_nonsupported_name
    }
}
