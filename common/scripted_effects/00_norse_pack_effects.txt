norse_pack_name_ai_male = {
	if = {
		limit = {
			NOT = {
				has_trait = bastard
			}
			exists = father
			father = {
				any_in_list = {
					variable = male_ancestors
					exists = this
				}
			}
		}
		father = {
			random_in_list = {
				variable = male_ancestors
				save_scope_as = male_ancestor
			}
		}
		if = {
			limit = {
				exists = dynasty
				dynasty = {
					any_in_list = {
						variable = name_pack_support_dynasty
						this = scope:male_ancestor
					}
				}
			}
			add_character_flag = name_pack_support
		}
		else_if = {
			limit = {
				any_in_global_list = {
					variable = name_pack_support_global
					this = scope:male_ancestor
				}
			}
			add_character_flag = name_pack_support
		}
		change_first_name = { template_character  = scope:male_ancestor }
		father = {
			remove_list_variable = { name = male_ancestors target = scope:male_ancestor }
		}
	}
	else_if = {
		limit = {
			NOT = {
				has_trait = bastard
			}
			exists = mother
			mother = {
				any_in_list = {
					variable = male_ancestors
					exists = this
				}
			}
		}
		father = {
			random_in_list = {
				variable = male_ancestors
				save_scope_as = male_ancestor
			}
		}
		if = {
			limit = {
				exists = dynasty
				dynasty = {
					any_in_list = {
						variable = name_pack_support_dynasty
						this = scope:male_ancestor
					}
				}
			}
			add_character_flag = name_pack_support
		}
		else_if = {
			limit = {
				any_in_global_list = {
					variable = name_pack_support_global
					this = scope:male_ancestor
				}
			}
			add_character_flag = name_pack_support
		}
		change_first_name = { template_character  = scope:male_ancestor }
		mother = {
			remove_list_variable = { name = male_ancestors target = scope:male_ancestor }
		}
	}
}
