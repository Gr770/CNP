courtier_effect_cnp = {
	if = {
		limit = {
			is_male = yes
			culture = {
				has_cultural_pillar = heritage_north_germanic
            	has_cultural_pillar = language_norse
			}
		}
		norse_pack_name_ai_male = yes
	}
	else_if = {
		limit = {
			is_male = no
			culture = {
				has_cultural_pillar = heritage_north_germanic
            	has_cultural_pillar = language_norse
			}
		}
		norse_pack_name_ai_female = yes
	}
	else_if = {
		limit = {
			is_male = yes
			culture = {
				is_cnp_supported_culture = yes
                has_cultural_pillar = language_arabic
			}
		}
		arabic_pack_name_ai_male = yes
	}
	else_if = {
		limit = {
			is_male = no
			culture = {
				is_cnp_supported_culture = yes
                has_cultural_pillar = language_arabic
			}
		}
		arabic_pack_name_ai_female = yes
	}
	else_if = {
		limit = {
			is_male = yes
			culture = {
            	has_cultural_pillar = heritage_iranian
			}
			has_religion = religion:christianity_religion
		}
		persian_christian_male_name_effect = yes
	}
	else_if = {
		limit = {
			is_male = yes
			culture = {
            	has_cultural_pillar = heritage_iranian
			}
			OR = {
				has_religion = religion:islam_religion
				culture = {
					has_cultural_pillar = language_arabic
				}
			}
		}
		persian_islam_male_name_effect = yes
	}
	else_if = {
		limit = {
			is_male = yes
			culture = {
            	has_cultural_pillar = heritage_iranian
			}
		}
		persian_fallback_male_name_effect = yes
	}
	else_if = {
		limit = {
			is_male = no
			culture = {
            	has_cultural_pillar = heritage_iranian
			}
			has_religion = religion:christianity_religion
		}
		persian_christian_female_name_effect = yes
	}
	else_if = {
		limit = {
			is_male = no
			culture = {
            	has_cultural_pillar = heritage_iranian
			}
			OR = {
				has_religion = religion:islam_religion
				culture = {
					has_cultural_pillar = language_arabic
				}
			}
		}
		persian_islam_female_name_effect = yes
	}
	else_if = {
		limit = {
			is_male = no
			culture = {
            	has_cultural_pillar = heritage_iranian
			}
		}
		persian_fallback_female_name_effect = yes
	}
}
